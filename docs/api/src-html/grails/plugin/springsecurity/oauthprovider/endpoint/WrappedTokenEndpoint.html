001package grails.plugin.springsecurity.oauthprovider.endpoint;
002
003import grails.plugin.springsecurity.oauthprovider.exceptions.OAuth2TokenEndpointException;
004import org.springframework.http.ResponseEntity;
005import org.springframework.security.oauth2.common.OAuth2AccessToken;
006import org.springframework.security.oauth2.common.exceptions.OAuth2Exception;
007import org.springframework.security.oauth2.provider.ClientRegistrationException;
008import org.springframework.security.oauth2.provider.endpoint.TokenEndpoint;
009import org.springframework.web.bind.annotation.RequestParam;
010
011import java.security.Principal;
012import java.util.Map;
013
014public class WrappedTokenEndpoint extends TokenEndpoint {
015
016    @Override
017    public ResponseEntity<OAuth2AccessToken> getAccessToken(Principal principal,
018            @RequestParam Map<String, String> parameters) {
019        try {
020            return super.getAccessToken(principal, parameters);
021        }
022        catch (ClientRegistrationException e) {
023            throw new OAuth2TokenEndpointException(e);
024        }
025        catch (OAuth2Exception e) {
026            throw new OAuth2TokenEndpointException(e);
027        }
028    }
029}
