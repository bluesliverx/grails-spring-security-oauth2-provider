001package grails.plugin.springsecurity.oauthprovider.endpoint;
002
003import grails.plugin.springsecurity.oauthprovider.exceptions.OAuth2AuthorizationEndpointException;
004import org.springframework.security.oauth2.common.exceptions.OAuth2Exception;
005import org.springframework.security.oauth2.provider.ClientRegistrationException;
006import org.springframework.security.oauth2.provider.endpoint.AuthorizationEndpoint;
007import org.springframework.web.HttpSessionRequiredException;
008import org.springframework.web.bind.annotation.RequestParam;
009import org.springframework.web.bind.support.SessionStatus;
010import org.springframework.web.servlet.ModelAndView;
011
012import java.security.Principal;
013import java.util.Map;
014
015public class WrappedAuthorizationEndpoint extends AuthorizationEndpoint {
016
017    @Override
018    public ModelAndView authorize(Map<String, Object> model,
019          @RequestParam(value = "response_type", required = false, defaultValue = "none") String responseType,
020          @RequestParam Map<String, String> requestParameters, SessionStatus sessionStatus, Principal principal) {
021        try {
022            return super.authorize(model, responseType, requestParameters, sessionStatus, principal);
023        }
024        catch (ClientRegistrationException e) {
025            throw new OAuth2AuthorizationEndpointException(e);
026        }
027        catch(OAuth2Exception e) {
028            throw new OAuth2AuthorizationEndpointException(e);
029        }
030    }
031}
